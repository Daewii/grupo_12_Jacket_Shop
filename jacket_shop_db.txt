//Creación de base de datos.

Create database jacket_shop_db;

//Creación de tablas
create table products (id int primary key auto_increment,
name varchar(50), 
description varchar(200), 
image varchar(200), 
price int,
materials_id int,
colors_id int,
sizes_id int);

create table category (id int primary key auto_increment, 
name varchar(50),
products_id int);

create table materials (id int primary key auto_increment, 
material varchar(500));

create table color (id int primary key auto_increment, 
name varchar(50));

create table users (id int primary key auto_increment, 
first_name varchar(50), 
last_name varchar(50), 
email varchar(50),
password varchar(50),
gender varchar(50),
profile_photo varchar(300));

//Relacionando tablas
ALTER TABLE categories  ADD CONSTRAINT products_id  
FOREIGN KEY (products_id)  
REFERENCES products (id)

ALTER TABLE products  ADD CONSTRAINT materials_id  
FOREIGN KEY (materials_id)  
REFERENCES materials (id)	

ALTER TABLE products  ADD CONSTRAINT colors_id  
FOREIGN KEY (colors_id)  
REFERENCES colors (id)	

ALTER TABLE products  ADD CONSTRAINT sizes_id  
FOREIGN KEY (sizes_id)  
REFERENCES sizes (id)	

alter table roles
add constraint users_id
foreign key(users_id)
references users (id);
