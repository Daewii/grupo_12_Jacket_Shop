Create database jacket_shop_db;
Use jacket_shop_db;
/*Creaci√≥n de tablas*/
create table products (id int primary key auto_increment,
name varchar(50), 
description varchar(200), 
image varchar(200), 
price int,
material_id int,
color_id int,
size_id int,
category_id int);

create table categories (id int primary key auto_increment, 
category varchar(50));

create table materials (id int primary key auto_increment, 
material varchar(500));

create table colors (id int primary key auto_increment, 
color varchar(50));

create table sizes (id int primary key auto_increment,
size varchar(50));

create table users (id int primary key auto_increment, 
first_name varchar(50), 
last_name varchar(50), 
email varchar(50),
password varchar(50),
profile_photo varchar(300),
rol_id int,
birthday date,
genre varchar(30)
);

create table roles (id int primary key auto_increment,
rol varchar(50));

/*Relacionando tablas*/
ALTER TABLE products  ADD CONSTRAINT category_id  
FOREIGN KEY (category_id  )  
REFERENCES categories (id);

ALTER TABLE products  ADD CONSTRAINT material_id  
FOREIGN KEY (material_id)  
REFERENCES materials (id);	

ALTER TABLE products  ADD CONSTRAINT color_id  
FOREIGN KEY (color_id)  
REFERENCES colors (id);

ALTER TABLE products  ADD CONSTRAINT size_id  
FOREIGN KEY (size_id)  
REFERENCES sizes (id);

alter table users
add constraint rol_id
foreign key(rol_id)
references roles (id);
